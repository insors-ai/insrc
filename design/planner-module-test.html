/**
 * Updates the status of a step in a plan, validating transitions and dependencies.
 * 
 * @param plan - The current plan object
 * @param stepId - ID of the step to update
 * @param newStatus - New status to assign
 * @param reason - Optional reason for the change
 * @returns Updated plan with modified step
 */
export function updateStepStatus<T>(
  plan: Plan<T>,
  stepId: string,
  newStatus: StepStatus,
  reason?: string
): Plan<T>;

/**
 * Validates that a status transition is allowed according to the defined state machine.
 * 
 * @param currentStatus - Current status of the step
 * @param newStatus - Target status
 * @returns True if valid, false otherwise
 */
export function isValidTransition(
  currentStatus: StepStatus,
  newStatus: StepStatus
): boolean;

/**
 * Validates that all dependencies of a step are in 'done' status before marking it as done.
 * 
 * @param plan - The current plan object
 * @param stepId - ID of the step to validate
 * @returns True if all dependencies are satisfied, false otherwise
 */
export function validateStepTransition<T>(
  plan: Plan<T>,
  stepId: string
): boolean;

/**
 * Detects steps that are blocked due to unmet dependencies.
 * 
 * @param plan - The current plan object
 * @returns List of blocked step IDs
 */
export function detectBlockedSteps<T>(plan: Plan<T>): string[];

/**
 * Logs a status change for a step.
 * 
 * @param stepId - ID of the step
 * @param historyEntry - History record to log
 * @returns Updated step with history appended
 */
export function logStatusChange(
  stepId: string,
  historyEntry: StepHistory
): void;

/**
 * Detects cycles in the dependency graph of a plan.
 * 
 * @param plan - The current plan object
 * @returns True if no cycle detected, false otherwise
 */
export function detectCycles<T>(plan: Plan<T>): boolean;

2. DATA FLOW

  1. Caller invokes updateStepStatus(plan, stepId, newStatus, reason?) in src/agent/planner/plan-engine.ts.
  2. updateStepStatus calls isValidTransition(currentStatus, newStatus) to verify the transition is legal per the allowed status graph (e.g., pending → in_progress, not pending → done directly). If invalid, throws an InvalidTransitionError.
  3. If newStatus === 'done', updateStepStatus calls validateStepTransition(plan, stepId) to confirm all steps listed in step.dependencies have status 'done'. If any dependency is not done, the transition is rejected with an UnmetDependenciesError.
  4. Before or during plan construction (not only at completion time), cycle detection is run across the dependencies graph of all steps using detectCycles(plan). This is a graph-level validation distinct from per-step dependency checking. It uses DFS traversal over Step.dependencies arrays. If a cycle is found, the plan mutation is rejected with a CircularDependencyError.
  5. On successful transition, updateStepStatus calls logStatusChange(stepId, historyEntry) recording { previousStatus, newStatus, timestamp, actor? } into StepHistory.
  6. The updated Plan<T> is returned immutably.

High-Risk Concerns

    • Risk: The local model included 'failed' and a truncated value in StepStatus.

    • Risk: Flat scan of steps misses nested dependency violations.
    • Mitigation: Implement recursive traversal in both validateStepTransition and detectCycles.

  4. Cycle Detection Scope:

    • Risk: Cycle detection only at plan creation, not during mutations.
    • Mitigation: Enforce immutability or validate dependencies on every change to step.dependencies.

Medium-Risk Concerns

  1. Error Handling Precision:

    • Risk: Generic Error throws reduce error handling precision.
    • Mitigation: Use typed error classes (CircularDependencyError, InvalidTransitionError, UnmetDependenciesError) for better consumer handling.

  2. Performance of DFS Cycle Detection:

    • Risk: DFS traversal on large deeply-nested plans may be slow.
    • Mitigation: Implement memoization or early termination strategies to optimize performance.

  3. Cross-Requirement Conflict on 'skipped' Status:

    • Risk: validateStepTransition might incorrectly block skipped steps from satisfying dependencies.
    • Mitigation: Explicitly handle 'skipped' as a valid terminal state in transition validation logic.

  4. Immutability of Dependencies:

    • Risk: If step dependencies are mutable, cycles can be introduced post-creation.
    • Mitigation: Enforce immutability or add validation on every dependency modification.

2. DATA FLOW

  1. Serialization (toMarkdown):

    • Input: Plan<T> (generic)
    • Process:
      • Extract metadata fields (title, author, created, updated) from plan
      • Generate YAML frontmatter block using these fields
      • Iterate over plan.steps recursively:
        • Render each step as a checkbox line (- [ ], - [x], etc.)
        • Embed hidden comment anchor (<!-- id: <step.id> -->) after checkbox
        • Indent sub-steps appropriately
    • Output: Complete Markdown string

  2. Deserialization (fromMarkdown):

    • Input: Raw Markdown string
    • Process:
      • Call parseFrontmatter() to extract metadata
      • Parse checkbox lines using regex:
        • Extract step ID from comment anchor (<!-- id: <step.id> -->)
        • Determine status from checkbox symbol
        • Reconstruct nested structure based on indentation depth
    • Output: Typed Plan<T> object

  3. Partial Update (updateStepInPlace):

    • Input: Markdown string, step ID, new status
    • Process:
      • Scan lines for matching comment anchor (<!-- id: <stepId> -->)
      • Locate the checkbox line and replace its symbol with corresponding one from newStatus
    • Output: Modified Markdown string

  4. Round-trip Fidelity:

    • fromMarkdown(toMarkdown(plan)) must reconstruct equivalent plan structure
    • Dependencies:
      • All StepStatus values mapped to valid checkbox symbols
      • Step IDs preserved in comment anchors
      • ISO date strings maintain fidelity through round-trip

2. DATA FLOW

  2. getProgressSummary(plan) in plan-engine.ts:

    • Calls collectStatusCounts(plan.steps, counts) to traverse and count statuses.
  3. collectStatusCounts(steps, counts) (private helper):
    • Iterates through each step in the input array.
    • Increments counts[step.status].
    • If step.subSteps exists and is non-empty:
      • Recursively calls itself with step.subSteps and the same counts object.
  4. After traversal, computes:
    • total = sum of all values in counts
    • percentComplete = (counts['done'] / total) * 100, guarded against division by zero.
  5. Returns a ProgressSummary object containing:
    • total: Total number of steps
    • byStatus: Counts per status
    • percentComplete: Percentage of completed steps
